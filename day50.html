1. pytest-asyncio

Python's standard pytest doesn't know how to run async def test functions. pytest-asyncio bridges this gap — essential when testing FastAPI, asyncio services, or any async code.

$ pip install pytest-asyncio
pytest.ini / pyproject.toml 
[tool.pytest.ini_options]
asyncio_mode = "auto"   # automatically handle async test functions
Python — async tests 
import asyncio
import pytest
import pytest_asyncio

# With asyncio_mode = "auto", just write async def test_*
async def test_async_function():
    result = await some_coroutine()
    assert result == "expected"

# Async fixtures
@pytest_asyncio.fixture
async def async_client():
    async with httpx.AsyncClient() as client:
        yield client
    # cleanup happens after each test

async def test_with_async_fixture(async_client):
    response = await async_client.get("https://httpbin.org/get")
    assert response.status_code == 200

# Testing asyncio.gather behaviour
async def test_concurrent_tasks():
    async def slow_op(n: int) -> int:
        await asyncio.sleep(0.01)
        return n * 2

    results = await asyncio.gather(slow_op(1), slow_op(2), slow_op(3))
    assert results == [2, 4, 6]

2. httpx AsyncClient as FastAPI Test Client

FastAPI's TestClient is synchronous (blocks the event loop). For testing async routes properly — especially SSE streaming or WebSocket endpoints — use httpx.AsyncClient with transport=ASGITransport.

$ pip install httpx pytest-asyncio
Python — conftest.py 
import pytest
import pytest_asyncio
import httpx
from httpx import AsyncClient, ASGITransport
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession
from sqlalchemy.orm import sessionmaker
from main import app
from database import Base, get_db

TEST_DB_URL = "sqlite+aiosqlite:///:memory:"

@pytest_asyncio.fixture(scope="session")
async def engine():
    engine = create_async_engine(TEST_DB_URL)
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
    yield engine
    await engine.dispose()

@pytest_asyncio.fixture
async def db(engine):
    async_session = sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)
    async with async_session() as session:
        yield session
        await session.rollback()   # clean state after each test

@pytest_asyncio.fixture
async def client(db):
    async def override_get_db():
        yield db
    app.dependency_overrides[get_db] = override_get_db
    async with AsyncClient(
        transport=ASGITransport(app=app),
        base_url="http://test",
    ) as ac:
        yield ac
    app.dependency_overrides.clear()
Python — async route tests 
async def test_create_task(client: AsyncClient):
    response = await client.post("/tasks/", json={"title": "Buy milk"})
    assert response.status_code == 201
    data = response.json()
    assert data["title"] == "Buy milk"

async def test_list_tasks(client: AsyncClient):
    await client.post("/tasks/", json={"title": "Task A"})
    await client.post("/tasks/", json={"title": "Task B"})
    response = await client.get("/tasks/")
    assert response.status_code == 200
    assert len(response.json()) >= 2

async def test_streaming_endpoint(client: AsyncClient):
    """Test SSE streaming — collect all chunks."""
    chunks = []
    async with client.stream("GET", "/chat/stream?prompt=hello") as response:
        async for line in response.aiter_lines():
            if line.startswith("data: ") and line != "data: [DONE]":
                chunks.append(line[6:])   # strip "data: " prefix
    assert len(chunks) > 0
    assert "".join(chunks)  # non-empty response

3. Testcontainers — Real DB in Tests

Testcontainers spins up real Docker containers for your tests — actual Postgres, actual Redis. Tests hit a real database with real SQL, catching issues that SQLite-in-memory never would.

$ pip install testcontainers
Python — Postgres Testcontainer 
import pytest
from testcontainers.postgres import PostgresContainer
from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker
from database import Base

@pytest.fixture(scope="session")
def postgres():
    """Start a real Postgres container for the test session."""
    with PostgresContainer("postgres:16-alpine") as pg:
        yield pg   # container is running; pg.get_connection_url() gives the DSN

@pytest.fixture(scope="session")
def engine(postgres):
    engine = create_engine(postgres.get_connection_url())
    Base.metadata.create_all(bind=engine)
    yield engine
    Base.metadata.drop_all(bind=engine)
    engine.dispose()

@pytest.fixture
def db(engine):
    connection = engine.connect()
    transaction = connection.begin()
    Session = sessionmaker(bind=connection)
    session = Session()
    yield session
    session.close()
    transaction.rollback()   # undo test changes
    connection.close()

def test_real_postgres(db):
    """This test hits a real Postgres instance."""
    result = db.execute(text("SELECT version()")).scalar()
    assert "PostgreSQL" in result

    "if TYPE_CHECKING:",
    "raise NotImplementedError",
    "@abstractmethod",

fail_under = 80

5. Factories for Test Data

Hardcoding test data in every test creates brittle tests. Factories produce realistic objects with sensible defaults that can be overridden per test.

$ pip install factory-boy
Python — factory_boy 
import factory
from factory.alchemy import SQLAlchemyModelFactory
from models import Task, TaskStatus
from database import SessionLocal

class TaskFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Task
        sqlalchemy_session = SessionLocal()
        sqlalchemy_session_persistence = "commit"

    title = factory.Sequence(lambda n: f"Task {n}")
    description = factory.Faker("sentence")
    status = TaskStatus.PENDING

# In tests:
def test_factory_creates_task(db):
    TaskFactory._meta.sqlalchemy_session = db
    task = TaskFactory()
    assert task.id is not None
    assert task.title.startswith("Task")

def test_multiple_tasks(db):
    TaskFactory._meta.sqlalchemy_session = db
    tasks = TaskFactory.create_batch(5, status=TaskStatus.DONE)
    assert len(tasks) == 5
    assert all(t.status == TaskStatus.DONE for t in tasks)

Lightweight Fixture Factories (No Library Needed)

Python — simple fixture factory 
import pytest
from tasklib.models import Task, Priority, Status

@pytest.fixture
def make_task():
    """Factory fixture — call with overrides per test."""
    _counter = {"n": 0}

    def _factory(
        title: str | None = None,
        priority: Priority = Priority.MEDIUM,
        status: Status = Status.PENDING,
    ) -> Task:
        _counter["n"] += 1
        return Task(
            id=_counter["n"],
            title=title or f"Task {_counter['n']}",
            priority=priority,
            status=status,
        )
    return _factory

# In tests:
def test_high_priority_filter(make_task):
    t1 = make_task(priority=Priority.HIGH)
    t2 = make_task(priority=Priority.LOW)
    high = [t for t in [t1, t2] if t.priority == Priority.HIGH]
    assert len(high) == 1
    assert high[0] is t1

Wrap-up

  • Set asyncio_mode = "auto" in pyproject.toml. Writing @pytest.mark.asyncio on every async test is mechanical noise — auto mode eliminates it project-wide.
  • Use httpx.AsyncClient with ASGITransport instead of FastAPI's TestClient when you need to test async behaviour properly — streaming, WebSockets, or background tasks that schedule coroutines.
  • Testcontainers tests are slower (Docker startup) but catch real bugs — constraint violations, index behaviour, isolation level issues — that SQLite-in-memory cannot. Run them in CI; skip in pre-commit hooks.
  • Coverage is a tool, not a goal. 80% meaningful coverage beats 100% coverage of trivial getters. Configure exclude_lines to skip @abstractmethod, TYPE_CHECKING blocks, and raise NotImplementedError.
  • Factory fixtures are better than hardcoded test data. A make_task() fixture lets each test say exactly what varies — everything else gets a sensible default.
