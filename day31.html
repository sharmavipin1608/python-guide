from sqlalchemy import create_engine
from sqlalchemy.orm import DeclarativeBase, sessionmaker, Session

# SQLite for development — swap DATABASE_URL for Postgres in production
DATABASE_URL = "sqlite:///./app.db"
# DATABASE_URL = "postgresql://user:pass@localhost/dbname"

engine = create_engine(
    DATABASE_URL,
    connect_args={"check_same_thread": False},  # SQLite-only setting
    echo=False,   # True = log all SQL (useful for debugging)
)

# Session factory — analogous to EntityManagerFactory
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

# Base class for all models
class Base(DeclarativeBase):
    pass

# FastAPI dependency — yields a session and closes it after the request
def get_db():
    db: Session = SessionLocal()
    try:
        yield db
    finally:
        db.close()

Defining Models

Java — JPA 
@Entity
@Table(name = "users")
public class User {
    @Id @GeneratedValue(strategy = AUTO)
    private Long id;

    @Column(nullable = false, unique = true)
    private String email;

    @Column(nullable = false)
    private String name;

Python — SQLAlchemy 2.x 
from datetime import datetime, timezone
from typing import Optional
from sqlalchemy import String, ForeignKey, func
from sqlalchemy.orm import Mapped, mapped_column, relationship
from database import Base

class User(Base):
    __tablename__ = "users"

    id:         Mapped[int]  = mapped_column(primary_key=True, index=True)
    email:      Mapped[str]  = mapped_column(String(255), unique=True, nullable=False, index=True)
    name:       Mapped[str]  = mapped_column(String(100), nullable=False)
    is_active:  Mapped[bool] = mapped_column(default=True)
    created_at: Mapped[datetime] = mapped_column(default=lambda: datetime.now(timezone.utc))

    # Relationship — loads posts lazily by default
    posts: Mapped[list["Post"]] = relationship(back_populates="author")

    def __repr__(self) -> str:
        return f"User(id={self.id}, email={self.email!r})"


class Post(Base):
    __tablename__ = "posts"

    id:         Mapped[int]          = mapped_column(primary_key=True, index=True)
    title:      Mapped[str]          = mapped_column(String(200), nullable=False)
    body:       Mapped[str]          = mapped_column(nullable=False)
    published:  Mapped[bool]         = mapped_column(default=False)
    user_id:    Mapped[int]          = mapped_column(ForeignKey("users.id"), nullable=False)
    created_at: Mapped[datetime]     = mapped_column(default=lambda: datetime.now(timezone.utc))

    author: Mapped["User"] = relationship(back_populates="posts")

CRUD Operations

Python 
from sqlalchemy.orm import Session
from models import User, Post

# ── CREATE ──────────────────────────────────────────────────────────────────

def create_user(db: Session, email: str, name: str) -> User:
    user = User(email=email, name=name)
    db.add(user)
    db.commit()        # writes to DB, assigns id
    db.refresh(user)   # reloads from DB (gets generated id, defaults)
    return user


# ── READ ─────────────────────────────────────────────────────────────────────

from sqlalchemy import select

def get_user(db: Session, user_id: int) -> User | None:
    return db.get(User, user_id)   # get by primary key — fastest path

def get_user_by_email(db: Session, email: str) -> User | None:
    stmt = select(User).where(User.email == email)
    return db.scalars(stmt).first()

def list_users(db: Session, skip: int = 0, limit: int = 100) -> list[User]:
    stmt = select(User).offset(skip).limit(limit).order_by(User.created_at.desc())
    return list(db.scalars(stmt))


# ── UPDATE ────────────────────────────────────────────────────────────────────

def update_user(db: Session, user_id: int, name: str) -> User | None:
    user = db.get(User, user_id)
    if not user:
        return None
    user.name = name   # modify tracked object
    db.commit()
    db.refresh(user)
    return user


# ── DELETE ────────────────────────────────────────────────────────────────────

def delete_user(db: Session, user_id: int) -> bool:
    user = db.get(User, user_id)
    if not user:
        return False
    db.delete(user)
    db.commit()
    return True

Python — SQLAlchemy 2.x 
from sqlalchemy import select, and_, or_, func

# Filter + order
stmt = (
    select(User)
    .where(User.is_active == True)
    .order_by(User.name)
)
users = list(db.scalars(stmt))

# Multiple conditions
stmt = select(User).where(
    and_(User.is_active == True, User.name.like("A%"))
)

# OR condition
stmt = select(User).where(
    or_(User.email.endswith("@gmail.com"), User.email.endswith("@yahoo.com"))
)

# Join
stmt = (
    select(User)
    .join(Post, Post.user_id == User.id)
    .where(Post.published == True)
    .distinct()
)

# Count
stmt = select(func.count(User.id)).where(User.is_active == True)
count = db.scalar(stmt)

# Aggregate
from sqlalchemy import func
stmt = (
    select(User.name, func.count(Post.id).label("post_count"))
    .join(Post, isouter=True)   # LEFT JOIN
    .group_by(User.id)
    .order_by(func.count(Post.id).desc())
)
for name, count in db.execute(stmt):
    print(f"{name}: {count} posts")

Relationships

One-to-many — User has many Posts

Python 
# Creating with relationship
user = create_user(db, "alice@example.com", "Alice")

post = Post(title="My First Post", body="Hello world!", user_id=user.id)
db.add(post)
db.commit()

# Accessing related objects
db.refresh(user)
print(user.posts)    # [Post(id=1, ...)]
print(post.author)   # User(id=1, email='alice@example.com')

# Eager loading — avoids N+1 queries
from sqlalchemy.orm import selectinload, joinedload

stmt = (
    select(User)
    .options(selectinload(User.posts))   # runs a second query for posts
    .where(User.is_active == True)
)
# OR use joinedload for a single SQL JOIN
stmt = (
    select(User)
    .options(joinedload(User.posts))
)

Many-to-many — Posts and Tags

Python 
from sqlalchemy import Table, Column, ForeignKey
from database import Base

# Association table — no model class needed for simple many-to-many
post_tags = Table(
    "post_tags",
    Base.metadata,
    Column("post_id", ForeignKey("posts.id"), primary_key=True),
    Column("tag_id",  ForeignKey("tags.id"),  primary_key=True),
)

class Tag(Base):
    __tablename__ = "tags"
    id:   Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50), unique=True)
    posts: Mapped[list["Post"]] = relationship(secondary=post_tags, back_populates="tags")

Alembic — Migrations

Shell 
alembic init alembic              # create alembic/ directory and alembic.ini

# Edit alembic/env.py — point it to your Base and DATABASE_URL:
# from models import Base
# target_metadata = Base.metadata
# config.set_main_option("sqlalchemy.url", DATABASE_URL)

alembic revision --autogenerate -m "create users and posts"   # detect model changes
alembic upgrade head              # apply all pending migrations
alembic downgrade -1              # roll back one migration
alembic history                   # list all migrations
Python — alembic/env.py key section 
import os
from models import Base   # import all your models so Alembic can detect them

# Point Alembic at your models' metadata
target_metadata = Base.metadata

# Use environment variable for the DB URL
config.set_main_option("sqlalchemy.url", os.environ["DATABASE_URL"])
Workflow: Change a model → alembic revision --autogenerate -m "describe change" → review the generated migration file → alembic upgrade head. Same mental model as Flyway's versioned scripts, but auto-generated from your model diff.

Wiring into FastAPI

Python — main.py 
from fastapi import FastAPI, Depends, HTTPException
from sqlalchemy.orm import Session
from typing import Annotated
from database import get_db, engine, Base
from models import User
from schemas import UserCreate, UserResponse
import crud

Base.metadata.create_all(bind=engine)   # creates tables on startup

app = FastAPI()
DB = Annotated[Session, Depends(get_db)]

@app.post("/users", response_model=UserResponse, status_code=201)
def create_user(body: UserCreate, db: DB):
    existing = crud.get_user_by_email(db, body.email)
    if existing:
        raise HTTPException(status_code=409, detail="Email already registered")
    return crud.create_user(db, email=body.email, name=body.name)

@app.get("/users/{user_id}", response_model=UserResponse)
def get_user(user_id: int, db: DB):
    user = crud.get_user(db, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    return user

@app.get("/users", response_model=list[UserResponse])
def list_users(skip: int = 0, limit: int = 100, db: DB = None):
    return crud.list_users(db, skip=skip, limit=limit)

Quick Reference Card

Engine
create_engine("sqlite:///./app.db")

One engine per app. For Postgres: "postgresql://user:pass@host/db"

Model field
id: Mapped[int] = mapped_column(primary_key=True)

SQLAlchemy 2.x style. Mapped[T] provides type hints.

Add and commit
db.add(obj); db.commit(); db.refresh(obj)

refresh() reloads from DB — gets generated id and server defaults.

Get by PK
db.get(User, user_id)

Fastest path for PK lookups. Returns None if not found.

Query
select(User).where(User.email == x)

Execute with db.scalars(stmt).first() or list(db.scalars(stmt)).

Eager load
select(User).options(selectinload(User.posts))

Avoids N+1. selectinload = separate query; joinedload = SQL JOIN.

Alembic autogenerate
alembic revision --autogenerate -m "add column"

Detects model changes and generates migration SQL. Always review before applying.

FastAPI session dep
DB = Annotated[Session, Depends(get_db)]

Yield-based — session opens on request, closes (with rollback on error) after.

Wrap-up

  • SQLAlchemy 2.x uses Mapped[T] type annotations for columns — this gives you IDE completion and mypy compatibility on your models, closer to how JPA annotations work in Java.
  • Always db.refresh(obj) after db.commit() when you need to read back server-generated values (auto-increment IDs, DEFAULT timestamps). Without it, the object's in-memory state may be stale.
  • Avoid the N+1 query problem: when you access a relationship in a loop, SQLAlchemy fires one SQL query per object. Use selectinload() or joinedload() to pre-fetch relationships in the initial query.
  • Alembic's --autogenerate compares your model definitions to the current database schema and generates migration scripts. Always review generated files before running — it doesn't catch every change (e.g., column rename vs drop+add).
  • The get_db dependency in FastAPI follows the yield pattern: the session is opened before the route runs and closed (or rolled back on exception) after. This is the standard pattern for request-scoped DB sessions.
