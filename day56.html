Option A — Real Estate Automation

Project Spec

  • What: An automated property listing monitor. Scrape listings on a schedule, store new ones in a DB, match against saved search criteria, send email alerts for matches.
  • Endpoints: POST /searches (save a search criterion), GET /listings (query stored listings), GET /listings/new (listings added in last 24h), POST /scrape (trigger a manual scrape)
  • Scraper: BeautifulSoup or Playwright — scrapes price, bedrooms, location, URL from a listing site
  • Background: Celery Beat triggers the scraper every hour; Celery worker sends email alerts on matches
  • Storage: SQLAlchemy + SQLite (or Postgres via docker-compose)
  • Covers: FastAPI, SQLAlchemy, BeautifulSoup (Day 53), Celery (Day 40), smtplib (Day 52), Docker (Day 33), config (Day 34)
Stretch Goal: Deploy to a VPS with Docker + nginx as a reverse proxy. Add a simple HTML front end (no framework needed — just Jinja2 templates served by FastAPI) to display listings and manage saved searches.

# tasks.py
from celery_app import celery_app
from scraper import scrape_all
from matcher import find_matches
from notifier import send_alert
import structlog

log = structlog.get_logger()

@celery_app.task(name="tasks.scrape_and_notify")
def scrape_and_notify() -> dict:
    new_listings = scrape_all()
    log.info("scrape_complete", new_listings=len(new_listings))
    matches = find_matches(new_listings)
    for match in matches:
        send_alert(match)
    return {"new": len(new_listings), "alerts_sent": len(matches)}

# celery_app.py — Beat schedule
celery_app.conf.beat_schedule = {
    "hourly-scrape": {
        "task": "tasks.scrape_and_notify",
        "schedule": 3600.0,   # every hour
    },
}

Option B — AI Dev Tool: Chat With Your Codebase

Project Spec

  • What: A FastAPI service that indexes Python source files into ChromaDB and answers questions about the codebase — with file path and approximate line number references in the answer.
  • Endpoints: POST /index (index a directory of .py files), GET /index/status (how many files/chunks indexed), POST /chat (ask a question, get answer + sources), DELETE /index (clear and re-index)

Python — AST-based code chunker 
import ast
from dataclasses import dataclass
from pathlib import Path

@dataclass
class CodeChunk:
    content: str
    file_path: str
    start_line: int
    end_line: int
    symbol_name: str    # function or class name
    symbol_type: str    # "function" or "class"

def chunk_python_file(filepath: Path) -> list[CodeChunk]:
    """Extract top-level functions and classes as chunks."""
    source = filepath.read_text(encoding="utf-8")
    try:
        tree = ast.parse(source)
    except SyntaxError:
        return []

    lines = source.splitlines()
    chunks = []

    for node in ast.walk(tree):
        if not isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
            continue
        # Only top-level definitions (depth == 1 in module)
        if not hasattr(node, "lineno"):
            continue

        start = node.lineno - 1
        end = node.end_lineno
        content = "\n".join(lines[start:end])

        chunks.append(CodeChunk(
            content=content,
            file_path=str(filepath),
            start_line=node.lineno,
            end_line=node.end_lineno,
            symbol_name=node.name,
            symbol_type="class" if isinstance(node, ast.ClassDef) else "function",
        ))

    # Also include module-level docstring if present
    if (isinstance(tree.body[0], ast.Expr)
            and isinstance(tree.body[0].value, ast.Constant)):
        docstring = tree.body[0].value.value
        chunks.insert(0, CodeChunk(
            content=f"# {filepath.name}\n{docstring}",
            file_path=str(filepath),
            start_line=1,
            end_line=tree.body[0].end_lineno,
            symbol_name="module_docstring",
            symbol_type="module",
        ))

    return chunks

def chunk_directory(root: Path, pattern: str = "**/*.py") -> list[CodeChunk]:
    all_chunks = []
    for filepath in root.rglob(pattern):
        if "__pycache__" in str(filepath):
            continue
        all_chunks.extend(chunk_python_file(filepath))
    return all_chunks
Python — index chunks into ChromaDB 
import chromadb
from chromadb.utils.embedding_functions import OpenAIEmbeddingFunction
from config import get_settings

def get_collection() -> chromadb.Collection:
    settings = get_settings()
    client = chromadb.PersistentClient(path=settings.chroma_path)
    embedding_fn = OpenAIEmbeddingFunction(
        api_key=settings.openai_api_key,
        model_name="text-embedding-3-small",
    )
    return client.get_or_create_collection("codebase", embedding_function=embedding_fn)

def index_chunks(chunks: list[CodeChunk]) -> int:
    collection = get_collection()
    collection.add(
        documents=[c.content for c in chunks],
        ids=[f"{c.file_path}:{c.start_line}" for c in chunks],
        metadatas=[{
            "file": c.file_path,
            "start_line": c.start_line,
            "end_line": c.end_line,
            "symbol": c.symbol_name,
            "type": c.symbol_type,
        } for c in chunks],
    )
    return len(chunks)
