Project: GitHub REST API Client

Build a real CLI tool that fetches GitHub user data, repositories, and open issues — then saves a full JSON report. Uses everything from the past week: requests, json, sys.argv, pathlib, datetime, and structured error handling.

Project Spec

What to build

  • A CLI tool: python github_client.py <username> [repo_name]
  • Without repo_name: show user profile + top 10 repos by stars
  • With repo_name: also show open issues for that repo (first page)
  • Save a full JSON report to report_<username>_<timestamp>.json
  • Print formatted output to terminal
  • Handle all errors gracefully: 404, network timeout, bad args

Stretch goals

  • Add pagination: fetch ALL repos using the Link header
  • Handle rate limiting: check X-RateLimit-Remaining header, sleep if near 0
  • Add a --token arg or read GITHUB_TOKEN env var for authenticated requests (higher rate limit)
  • Use ThreadPoolExecutor to fetch user + repos + issues concurrently

"""
GitHub REST API Client
Usage: python github_client.py <username> [repo_name]
Env:   GITHUB_TOKEN — optional personal access token
"""
import json
import os
import sys
from datetime import datetime, timezone
from pathlib import Path

import requests
from requests.exceptions import HTTPError, Timeout, ConnectionError


# ── Configuration ────────────────────────────────────────────────────────────

BASE_URL = "https://api.github.com"
TIMEOUT  = 10   # seconds

def build_session() -> requests.Session:
    session = requests.Session()
    session.headers.update({
        "Accept": "application/vnd.github.v3+json",
        "User-Agent": "python-github-client/1.0",
    })
    token = os.environ.get("GITHUB_TOKEN")
    if token:
        session.headers["Authorization"] = f"Bearer {token}"
        print("[auth] Using GITHUB_TOKEN")
    return session


# ── API calls ─────────────────────────────────────────────────────────────────

def get_user(session: requests.Session, username: str) -> dict:
    """Fetch a GitHub user profile. Raises ValueError if not found."""
    resp = session.get(f"{BASE_URL}/users/{username}", timeout=TIMEOUT)
    try:
        resp.raise_for_status()
    except HTTPError as e:
        if resp.status_code == 404:
            raise ValueError(f"User '{username}' not found on GitHub.") from e
        raise
    return resp.json()


def get_repos(session: requests.Session, username: str,
              per_page: int = 100) -> list[dict]:
    """Fetch all public repos for a user, sorted by stars."""
    repos = []
    page = 1
    while True:
        resp = session.get(
            f"{BASE_URL}/users/{username}/repos",
            params={"per_page": per_page, "page": page, "sort": "updated"},
            timeout=TIMEOUT,
        )
        resp.raise_for_status()
        batch = resp.json()
        if not batch:
            break
        repos.extend(batch)

        # Check for next page via Link header
        if 'next' not in resp.links:
            break
        page += 1

    return sorted(repos, key=lambda r: r["stargazers_count"], reverse=True)


def get_issues(session: requests.Session, username: str,
               repo: str, per_page: int = 30) -> list[dict]:
    """Fetch open issues for a repo (excludes pull requests)."""
    resp = session.get(
        f"{BASE_URL}/repos/{username}/{repo}/issues",
        params={"state": "open", "per_page": per_page},
        timeout=TIMEOUT,
    )
    try:
        resp.raise_for_status()
    except HTTPError as e:
        if resp.status_code == 404:
            raise ValueError(f"Repo '{username}/{repo}' not found.") from e
        raise
    # GitHub's /issues endpoint includes pull requests — filter them out
    return [i for i in resp.json() if "pull_request" not in i]


# ── Output formatting ─────────────────────────────────────────────────────────

def print_user(user: dict) -> None:
    print(f"\n{'='*50}")
    print(f"  {user['login']}  ({user.get('name', 'No name')})")
    print(f"{'='*50}")
    if user.get("bio"):
        print(f"  Bio:       {user['bio']}")
    print(f"  Location:  {user.get('location') or '—'}")
    print(f"  Company:   {user.get('company') or '—'}")
    print(f"  Followers: {user['followers']:,}   Following: {user['following']:,}")
    print(f"  Public repos: {user['public_repos']}")
    if user.get("blog"):
        print(f"  Website:   {user['blog']}")


def print_repos(repos: list[dict], limit: int = 10) -> None:
    print(f"\n  Top {min(limit, len(repos))} repositories by stars:\n")
    print(f"  {'Repository':<35} {'Stars':>7} {'Forks':>6}  Language")
    print(f"  {'-'*35} {'-'*7} {'-'*6}  {'-'*12}")
    for r in repos[:limit]:
        lang = r.get("language") or "—"
        print(f"  {r['name']:<35} {r['stargazers_count']:>7,} {r['forks_count']:>6,}  {lang}")


def print_issues(issues: list[dict], repo: str, limit: int = 10) -> None:
    if not issues:
        print(f"\n  No open issues in {repo}.")
        return
    print(f"\n  Open issues in {repo} ({len(issues)} shown):\n")
    for i in issues[:limit]:
        labels = ", ".join(l["name"] for l in i.get("labels", []))
        label_str = f"  [{labels}]" if labels else ""
        print(f"  #{i['number']:<6} {i['title'][:60]}{label_str}")


# ── Report generation ──────────────────────────────────────────────────────────

def save_report(username: str, data: dict) -> Path:
    ts   = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    path = Path(f"report_{username}_{ts}.json")
    path.write_text(json.dumps(data, indent=2, default=str))
    return path


# ── Entry point ────────────────────────────────────────────────────────────────

def main() -> None:
    args = sys.argv[1:]
    if not args:
        print("Usage: python github_client.py <username> [repo_name]", file=sys.stderr)
        sys.exit(1)

    username  = args[0]
    repo_name = args[1] if len(args) > 1 else None

    session = build_session()
    report: dict = {"generated_at": datetime.now(timezone.utc).isoformat(),
                    "username": username}

    try:
        # Always fetch user profile
        user = get_user(session, username)
        report["user"] = user
        print_user(user)

        # Always fetch repos
        repos = get_repos(session, username)
        report["repos"] = repos
        print_repos(repos)

        # Optionally fetch issues
        if repo_name:
            issues = get_issues(session, username, repo_name)
            report["issues"] = {"repo": repo_name, "items": issues}
            print_issues(issues, repo_name)

    except ValueError as e:
        print(f"\nError: {e}", file=sys.stderr)
        sys.exit(1)
    except Timeout:
        print("\nError: GitHub API timed out. Try again.", file=sys.stderr)
        sys.exit(1)
    except ConnectionError:
        print("\nError: Cannot reach GitHub API. Check your connection.", file=sys.stderr)
        sys.exit(1)
    except HTTPError as e:
        print(f"\nHTTP Error: {e}", file=sys.stderr)
        sys.exit(1)

    # Save report
    path = save_report(username, report)
    print(f"\n  Report saved → {path}")


if __name__ == "__main__":
    main()

Stretch: Concurrent fetch with ThreadPoolExecutor

Python 
from concurrent.futures import ThreadPoolExecutor, as_completed

def fetch_all_concurrent(session, username, repo_name=None):
    """Fetch user, repos, and issues concurrently."""
    with ThreadPoolExecutor(max_workers=3) as pool:
        futures = {
            "user":  pool.submit(get_user,  session, username),
            "repos": pool.submit(get_repos, session, username),
        }
        if repo_name:
            futures["issues"] = pool.submit(
                get_issues, session, username, repo_name
            )

        results = {}
        for key, future in futures.items():
            try:
                results[key] = future.result(timeout=30)
            except Exception as e:
                print(f"Failed to fetch {key}: {e}", file=sys.stderr)
                results[key] = None

    return results

Stretch: Rate limit handling

Python 
import time

def checked_get(session: requests.Session, url: str, **kwargs) -> requests.Response:
    """Make a GET request and handle GitHub rate limiting."""
    resp = session.get(url, **kwargs)

    remaining = int(resp.headers.get("X-RateLimit-Remaining", 999))
    reset_at   = int(resp.headers.get("X-RateLimit-Reset", 0))

    if remaining < 5:
        wait = max(0, reset_at - int(time.time())) + 1
        print(f"Rate limit nearly exhausted. Waiting {wait}s...", file=sys.stderr)
        time.sleep(wait)

    if resp.status_code == 403 and "rate limit" in resp.text.lower():
        print("Rate limited. Waiting 60s...", file=sys.stderr)
        time.sleep(60)
        resp = session.get(url, **kwargs)

    return resp

Quick Reference Card

Session with auth
s.headers["Authorization"] = f"Bearer {token}"

Set once on session — applies to all requests.

Pagination via Link header
if 'next' not in resp.links: break

requests parses the Link header into resp.links dict.

Save JSON report
path.write_text(json.dumps(data, indent=2, default=str))

default=str converts datetime etc. to strings as fallback.

Timestamped filename
datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

Always use UTC in filenames — avoids timezone confusion.

Rate limit headers
resp.headers.get("X-RateLimit-Remaining")

GitHub returns rate limit info on every response. Check before hammering.

Filter PRs from issues
[i for i in data if "pull_request" not in i]

GitHub's /issues endpoint includes PRs. Filter by absence of "pull_request" key.

Wrap-up

  • A requests.Session is the right abstraction for an API client — set auth headers and base config once, then every method call inherits them without repetition.
  • Pagination is handled by following the Link: <url>; rel="next" header. requests parses this for you into resp.links["next"]["url"] — always use it instead of guessing page counts.
  • Always catch HTTPError, Timeout, and ConnectionError separately. Each needs a different response: 404 = user error, timeout = retry, connection failure = infrastructure problem.
  • The default=str kwarg in json.dumps is a quick way to make reports serializable even when data contains datetime or other non-JSON types — fine for reports, not for APIs.
  • Threading concurrency (stretch goal) provides a real speedup here because all three calls are I/O-bound — user, repos, and issues can be fetched simultaneously with ThreadPoolExecutor.
