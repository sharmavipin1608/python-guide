Dependency Injection with Depends()

FastAPI's Depends() is constructor injection for route handlers. Declare a dependency function, and FastAPI calls it automatically — injecting its return value into your route. Dependencies can depend on other dependencies, forming a tree.

Java — Spring 
@RestController
public class ItemController {
    @Autowired
    private ItemRepository repo;    // injected by Spring

    @GetMapping("/items")
    public List<Item> list() {
        return repo.findAll();
    }
}
Python — FastAPI 
from fastapi import FastAPI, Depends
from typing import Annotated

app = FastAPI()

# Dependency function — can be anything callable
def get_db():
    db = connect_to_db()   # fake connection
    try:
        yield db           # yield = cleanup after request
    finally:
        db.close()

# Type alias for cleaner signatures
DB = Annotated[Database, Depends(get_db)]

@app.get("/items")
def list_items(db: DB):
    return db.query("SELECT * FROM items")

@app.post("/items")
def create_item(body: ItemCreate, db: DB):
    return db.insert(body.model_dump())

Auth dependency — reusable across routes

Python 
from fastapi import Depends, Header, HTTPException
from typing import Annotated

# Shared auth dependency
def verify_token(x_api_key: str = Header()) -> str:
    """Extract and validate API key from header."""
    if x_api_key != "secret-key":
        raise HTTPException(status_code=401, detail="Invalid API key")
    return x_api_key

# Type alias — use Annotated for clean signatures
AuthToken = Annotated[str, Depends(verify_token)]

@app.get("/protected")
def protected_route(token: AuthToken):
    return {"message": "Access granted", "token": token}

# Apply to all routes in a router
from fastapi import APIRouter
router = APIRouter(dependencies=[Depends(verify_token)])

@router.get("/admin/users")
def list_users():
    return []   # token already verified by the router-level dependency

Dependency with state

Python 
from fastapi import Depends
from functools import lru_cache

class Settings:
    db_url: str = "sqlite:///./test.db"
    debug: bool = False

@lru_cache   # called once per process — acts like a singleton
def get_settings() -> Settings:
    return Settings()

SettingsDep = Annotated[Settings, Depends(get_settings)]

@app.get("/config")
def show_config(settings: SettingsDep):
    return {"db_url": settings.db_url, "debug": settings.debug}

Middleware

Middleware intercepts every request before it reaches a route and every response before it leaves. Use it for logging, timing, authentication, compression, or custom headers.

Python 
import time
import uuid
from fastapi import FastAPI, Request, Response
from starlette.middleware.base import BaseHTTPMiddleware

app = FastAPI()

# Middleware as a class (more control)
class TimingMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        start = time.perf_counter()
        response: Response = await call_next(request)   # call the actual route
        elapsed = time.perf_counter() - start
        response.headers["X-Process-Time"] = f"{elapsed:.4f}s"
        return response

class RequestIdMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        request_id = str(uuid.uuid4())
        # Attach to request state for use in route handlers
        request.state.request_id = request_id
        response = await call_next(request)
        response.headers["X-Request-ID"] = request_id
        return response

app.add_middleware(TimingMiddleware)
app.add_middleware(RequestIdMiddleware)

# Middleware as a decorator — simpler for one-off cases
@app.middleware("http")
async def log_requests(request: Request, call_next):
    print(f"→ {request.method} {request.url.path}")
    response = await call_next(request)
    print(f"← {response.status_code}")
    return response

Background Tasks

Background tasks run after the response is sent to the client. Use them for operations that shouldn't make the client wait: sending email, writing audit logs, triggering async jobs.

Java — @Async 
@Service
public class NotificationService {
    @Async
    public CompletableFuture<Void> sendEmail(String to, String msg) {
        // runs in background thread after response sent
        emailClient.send(to, msg);
        return CompletableFuture.completedFuture(null);
    }
}
Python 
from fastapi import FastAPI, BackgroundTasks
import smtplib

app = FastAPI()

def send_welcome_email(email: str, name: str) -> None:
    """Runs after the response is sent — doesn't block the client."""
    print(f"Sending welcome email to {email}...")
    # smtplib.sendmail(...) in reality
    import time; time.sleep(2)   # simulate slow email
    print(f"Email sent to {email}")

@app.post("/users", status_code=201)
def create_user(body: UserCreate, background_tasks: BackgroundTasks):
    user = save_user(body)
    # Queue the email — will run AFTER this function returns
    background_tasks.add_task(send_welcome_email, body.email, body.name)
    return user   # client gets response immediately

Custom Exception Handlers

Python 
from fastapi import FastAPI, Request, HTTPException
from fastapi.responses import JSONResponse

app = FastAPI()

# Custom exception class
class ItemNotFoundError(Exception):
    def __init__(self, item_id: int):
        self.item_id = item_id

# Register a handler — like @ControllerAdvice in Spring
@app.exception_handler(ItemNotFoundError)
async def item_not_found_handler(request: Request, exc: ItemNotFoundError):
    return JSONResponse(
        status_code=404,
        content={
            "error": "NOT_FOUND",
            "message": f"Item {exc.item_id} does not exist",
            "path": str(request.url.path),
        },
    )

# Override the default 422 validation error response
from fastapi.exceptions import RequestValidationError

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=422,
        content={
            "error": "VALIDATION_ERROR",
            "fields": exc.errors(),
        },
    )

@app.get("/items/{item_id}")
def get_item(item_id: int):
    if item_id not in db:
        raise ItemNotFoundError(item_id)   # custom exception
    return db[item_id]

Routers — Splitting Large Apps

As an app grows, putting everything in main.py becomes unmanageable. APIRouter lets you group routes by domain — like Spring's @Controller per package.

Python — routers/users.py 
from fastapi import APIRouter, Depends, HTTPException

router = APIRouter(
    prefix="/users",            # all routes here start with /users
    tags=["users"],             # groups in /docs Swagger UI
    dependencies=[Depends(verify_token)],  # applied to all routes in this router
)

@router.get("/")
def list_users(): ...

@router.get("/{user_id}")
def get_user(user_id: int): ...

@router.post("/", status_code=201)
def create_user(body: UserCreate): ...
Python — routers/items.py 
from fastapi import APIRouter

router = APIRouter(prefix="/items", tags=["items"])

@router.get("/")
def list_items(): ...

@router.post("/", status_code=201)
def create_item(body: ItemCreate): ...
Python — main.py 
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from routers import users, items

app = FastAPI(title="My API", version="2.0")

# CORS
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])

# Mount routers
app.include_router(users.router)
app.include_router(items.router)

# Health check — lives in main
@app.get("/health", tags=["system"])
def health(): return {"status": "ok"}

Quick Reference Card

Yield-based dependencies auto-clean up after the request. Used for DB connections.

Inject dependency

Type alias + Annotated keeps signatures clean. FastAPI resolves automatically.

Middleware class

CORS

Required for browser-based frontends hitting a different origin/port.

Exception handler

Like Spring's @ControllerAdvice. Centralises error response format.

Applies a dependency to all routes in the router — no need to repeat on each.

Wrap-up

  • Depends() is FastAPI's dependency injection — declare a function, and FastAPI calls it and injects the result. Use yield-based dependencies for resources that need cleanup (DB connections, file handles). Dependencies compose: a dependency can depend on other dependencies.
  • Middleware runs on every request/response — use it sparingly and keep it fast. Add timing, request IDs, and logging there. Business logic belongs in route handlers or services, not middleware.
  • BackgroundTasks is the simple, no-install solution for fire-and-forget work (emails, webhooks, audit logs). For heavier workloads that need retries, scheduling, or distributed execution, use Celery (Day 40).
  • Custom exception handlers centralise your error response format. Raise domain-specific exceptions in your route handlers and let the handlers format the response — same pattern as Spring's @ControllerAdvice.
  • APIRouter is how you scale a FastAPI app. One router per domain, each with its own prefix, tags, and shared dependencies. Your main.py becomes a thin wiring file — just include_router calls and app-level config.
