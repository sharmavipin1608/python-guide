Project: Sales Data Pipeline

Read raw messy CSV data, clean it with Pandas, analyze it across multiple dimensions, and output a clean data file plus a printed summary report. This is the core of every data engineering task.

Project Spec

What to build

  • Read raw_sales.csv (sample data provided below)
  • Clean: handle nulls, fix types, normalize column names, remove duplicates
  • Analyze: total revenue by region, top 5 products by revenue, monthly trend
  • Output: clean_sales.csv + a formatted terminal summary report
  • All in a single pipeline.py script with clear functions for each stage

Stretch goals

  • Add a chart using matplotlib: bar chart of revenue by region
  • Add a --input CLI argument so the script works on any CSV
  • Write a pytest test suite for the clean() and analyze() functions
  • Export the summary to an Excel file with multiple sheets

"""
Sales Data Pipeline
Usage: python pipeline.py [--input raw_sales.csv]
"""
import sys
from pathlib import Path

import numpy as np
import pandas as pd


# ── 1. Load ───────────────────────────────────────────────────────────────────

def load(path: str | Path) -> pd.DataFrame:
    """Load raw CSV with appropriate types."""
    df = pd.read_csv(
        path,
        parse_dates=["Date"],
        dtype={"Units": float, "Unit_Price": float},
    )
    print(f"Loaded {len(df):,} rows from {path}")
    print(f"Missing values:\n{df.isna().sum()[df.isna().sum() > 0]}\n")
    return df


# ── 2. Clean ──────────────────────────────────────────────────────────────────

def clean(df: pd.DataFrame) -> pd.DataFrame:
    """Normalize, fix types, handle nulls, deduplicate."""
    result = df.copy()

    # Normalize column names: lowercase, underscores
    result.columns = result.columns.str.lower().str.replace(" ", "_")

    # Normalize string columns to title case
    result["region"]  = result["region"].str.strip().str.title()
    result["product"] = result["product"].str.strip()
    result["rep"]     = result["rep"].str.strip()

    # Fill missing units with median of same product
    for product in result["product"].unique():
        mask = result["product"] == product
        median_units = result.loc[mask, "units"].median()
        result.loc[mask & result["units"].isna(), "units"] = median_units

    # Cast units to int now that nulls are filled
    result["units"] = result["units"].astype(int)

    # Derived column: total revenue per row
    result["revenue"] = result["units"] * result["unit_price"]

    # Add time columns for analysis
    result["month"]   = result["date"].dt.to_period("M").astype(str)
    result["quarter"] = result["date"].dt.to_period("Q").astype(str)

    # Drop true duplicates (same date + region + product + rep)
    before = len(result)
    result = result.drop_duplicates(subset=["date", "region", "product", "rep"])
    dropped = before - len(result)
    if dropped:
        print(f"Dropped {dropped} duplicate rows")

    print(f"Clean dataset: {len(result):,} rows, {result['revenue'].sum():,.2f} total revenue\n")
    return result


# ── 3. Analyze ────────────────────────────────────────────────────────────────

def analyze(df: pd.DataFrame) -> dict:
    """Compute summary statistics. Returns dict of DataFrames."""

    # Revenue by region
    by_region = (
        df.groupby("region")
        .agg(
            total_revenue=("revenue", "sum"),
            total_units=("units",   "sum"),
            transactions=("revenue", "count"),
        )
        .sort_values("total_revenue", ascending=False)
        .round(2)
    )

    # Top 5 products by revenue
    top_products = (
        df.groupby("product")
        .agg(
            total_revenue=("revenue", "sum"),
            total_units=("units",   "sum"),
            avg_price=("unit_price", "mean"),
        )
        .sort_values("total_revenue", ascending=False)
        .head(5)
        .round(2)
    )

    # Monthly revenue trend
    monthly = (
        df.groupby("month")["revenue"]
        .sum()
        .reset_index()
        .rename(columns={"revenue": "monthly_revenue"})
        .round(2)
    )
    # Add month-over-month growth %
    monthly["mom_growth"] = monthly["monthly_revenue"].pct_change() * 100
    monthly["mom_growth"] = monthly["mom_growth"].round(1)

    # Rep performance
    by_rep = (
        df.groupby("rep")["revenue"]
        .agg(["sum", "mean", "count"])
        .rename(columns={"sum": "total", "mean": "avg_per_sale", "count": "sales"})
        .sort_values("total", ascending=False)
        .round(2)
    )

    return {
        "by_region":    by_region,
        "top_products": top_products,
        "monthly":      monthly,
        "by_rep":       by_rep,
    }


# ── 4. Report ─────────────────────────────────────────────────────────────────

def print_report(df: pd.DataFrame, stats: dict) -> None:
    """Print a formatted terminal summary."""
    total_rev  = df["revenue"].sum()
    total_units = df["units"].sum()
    date_range  = f"{df['date'].min().date()} → {df['date'].max().date()}"

    print("=" * 60)
    print("  SALES SUMMARY REPORT")
    print("=" * 60)
    print(f"  Period:        {date_range}")
    print(f"  Total Revenue: ${total_rev:>12,.2f}")
    print(f"  Total Units:   {total_units:>12,}")
    print(f"  Transactions:  {len(df):>12,}")

    print("\n── Revenue by Region ──────────────────────────────────")
    for region, row in stats["by_region"].iterrows():
        bar = "█" * int(row["total_revenue"] / total_rev * 30)
        print(f"  {region:<10} ${row['total_revenue']:>8,.0f}  {bar}")

    print("\n── Top Products ───────────────────────────────────────")
    print(f"  {'Product':<12} {'Revenue':>10}  {'Units':>7}  {'Avg Price':>10}")
    print(f"  {'-'*12} {'-'*10}  {'-'*7}  {'-'*10}")
    for prod, row in stats["top_products"].iterrows():
        print(f"  {prod:<12} ${row['total_revenue']:>9,.2f}  {row['total_units']:>7,}  ${row['avg_price']:>9.2f}")

    print("\n── Monthly Revenue Trend ──────────────────────────────")
    for _, row in stats["monthly"].iterrows():
        growth = f"({row['mom_growth']:+.1f}%)" if pd.notna(row["mom_growth"]) else ""
        print(f"  {row['month']}   ${row['monthly_revenue']:>8,.2f}  {growth}")

    print("\n── Rep Performance ────────────────────────────────────")
    for rep, row in stats["by_rep"].iterrows():
        print(f"  {rep:<8} ${row['total']:>8,.2f}  ({row['sales']} sales, avg ${row['avg_per_sale']:,.2f})")

    print("\n" + "=" * 60)


# ── 5. Export ─────────────────────────────────────────────────────────────────

def export(df: pd.DataFrame, stats: dict, output_dir: Path) -> None:
    """Write clean CSV and summary sheets to disk."""
    output_dir.mkdir(exist_ok=True)

    clean_path = output_dir / "clean_sales.csv"
    df.to_csv(clean_path, index=False)
    print(f"\nClean data → {clean_path}")

    summary_path = output_dir / "summary.xlsx"
    with pd.ExcelWriter(summary_path) as writer:
        for sheet_name, frame in stats.items():
            frame.to_excel(writer, sheet_name=sheet_name)
    print(f"Summary    → {summary_path}")


# ── Entry point ───────────────────────────────────────────────────────────────

def main() -> None:
    input_path = Path(sys.argv[1]) if len(sys.argv) > 1 else Path("raw_sales.csv")

    if not input_path.exists():
        print(f"Error: {input_path} not found", file=sys.stderr)
        sys.exit(1)

    raw      = load(input_path)
    clean_df = clean(raw)
    stats    = analyze(clean_df)
    print_report(clean_df, stats)
    export(clean_df, stats, Path("output"))


if __name__ == "__main__":
    main()

Key Pipeline Patterns

Stage functions — pure transformations

Each stage (load, clean, analyze) takes a DataFrame and returns a DataFrame. They're pure: no global state, no side effects except logging. This makes each stage independently testable.

Java parallel: This is the same pipeline pattern you'd use in Spring Batch — Reader → Processor → Writer. Each step is a pure transformation function, easy to test and easy to swap out.

import matplotlib.pyplot as plt
import matplotlib.ticker as mticker

def plot_revenue_by_region(by_region: pd.DataFrame, output_dir: Path) -> None:
    """Save a bar chart of revenue by region."""
    fig, ax = plt.subplots(figsize=(8, 4))
    fig.patch.set_facecolor("#0d0d0d")
    ax.set_facecolor("#141414")

    regions = by_region.index
    revenues = by_region["total_revenue"]
    bars = ax.bar(regions, revenues, color="#f0e040", width=0.6)

    ax.set_title("Revenue by Region", color="#f0e040", fontsize=14, pad=16)
    ax.set_xlabel("Region", color="#888")
    ax.set_ylabel("Revenue ($)", color="#888")
    ax.tick_params(colors="#888")
    ax.yaxis.set_major_formatter(mticker.FuncFormatter(lambda x, _: f"${x:,.0f}"))
    for spine in ax.spines.values():
        spine.set_edgecolor("#2a2a2a")

    path = output_dir / "revenue_by_region.png"
    fig.savefig(path, dpi=150, bbox_inches="tight", facecolor="#0d0d0d")
    plt.close(fig)
    print(f"Chart      → {path}")

Quick Reference Card

Read with type hints
pd.read_csv(f, parse_dates=["date"], dtype={"amt": float})

Parse dates and enforce dtypes at load time — cheaper than fixing later.

Fill nulls by group
df.loc[mask & df["col"].isna(), "col"] = df.loc[mask, "col"].median()

Fill missing values with the group's own median, not the overall median.

Derived column
df["revenue"] = df["units"] * df["price"]

Vectorized — no loop. Result is a new column.

Multi-agg groupby
df.groupby("col").agg(total=("rev","sum"), count=("rev","count"))

Named aggregations keep the output column names clean.

pct_change
series.pct_change() * 100

Month-over-month or period-over-period growth %. First value is NaN.

Multi-sheet Excel

Each DataFrame goes to a separate sheet. Requires openpyxl.

Wrap-up

  • Separate your pipeline into pure stage functions: load → clean → analyze → report → export. Each takes a DataFrame and returns one — independently testable, composable, and easy to debug by inspecting intermediate outputs.
  • Fix data quality at the boundary — normalize casing, fill nulls, fix types, and drop duplicates in clean(). Downstream analysis code should be able to trust that the data is valid.
  • Derived columns (revenue = units * price) are free to add — vectorized, no loop. Add them in the clean stage so they're available everywhere downstream.
  • Named aggregations (total_revenue=("revenue", "sum")) give you clean output column names in one call — much cleaner than renaming after the fact.
  • This pipeline pattern — load, clean, analyze, export — is the skeleton of every data engineering job, Airflow DAG, and data science notebook you'll encounter. The tools change; the structure stays the same.
