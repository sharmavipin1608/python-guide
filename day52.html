1. subprocess — Run Shell Commands

subprocess.run() is the modern way to run external programs from Python. It replaces the old os.system() and gives you full control over stdin, stdout, stderr, and return codes.

Python — subprocess basics 
import subprocess

# Simple command — capture output
result = subprocess.run(
    ["git", "status"],
    capture_output=True,    # captures stdout and stderr
    text=True,              # decode bytes to str automatically
    check=True,             # raise CalledProcessError if exit code != 0
)
print(result.stdout)
print(result.returncode)   # 0 = success

# Run a shell pipeline (use shell=True carefully — avoid with user input)
result = subprocess.run(
    "ls -la | grep .py",
    shell=True,
    capture_output=True,
    text=True,
)

# Pass input to stdin
result = subprocess.run(
    ["sort"],
    input="banana\napple\ncherry\n",
    capture_output=True,
    text=True,
)
print(result.stdout)   # apple\nbanana\ncherry\n

Practical Automation Helpers

Python 
import subprocess
from pathlib import Path

def run(cmd: list[str], cwd: Path | None = None) -> str:
    """Run a command, raise on failure, return stdout."""
    result = subprocess.run(
        cmd, capture_output=True, text=True, check=True, cwd=cwd
    )
    return result.stdout.strip()

def git_current_branch() -> str:
    return run(["git", "rev-parse", "--abbrev-ref", "HEAD"])

def git_log(n: int = 10) -> list[str]:
    output = run(["git", "log", f"--oneline", f"-{n}"])
    return output.splitlines()

def docker_ps() -> list[dict]:
    """List running containers as structured data."""
    output = run(["docker", "ps", "--format", "{{.ID}}\t{{.Image}}\t{{.Status}}"])
    containers = []
    for line in output.splitlines():
        parts = line.split("\t")
        containers.append({"id": parts[0], "image": parts[1], "status": parts[2]})
    return containers

# Run multiple commands, abort on first failure
def deploy(tag: str) -> None:
    run(["docker", "build", "-t", f"myapp:{tag}", "."])
    run(["docker", "push", f"myapp:{tag}"])
    run(["kubectl", "set", "image", "deployment/myapp", f"app=myapp:{tag}"])
    print(f"Deployed myapp:{tag}")
Security warning: Never use shell=True with user-provided input — it enables shell injection. Pass commands as a list (["git", "status"]), not a string, whenever possible.

2. shutil — File Operations

shutil (shell utilities) handles high-level file operations that pathlib and os don't cover: copying with metadata, moving across filesystems, and creating archives.

Python 
import shutil
from pathlib import Path

src = Path("reports/2026-04")
dst = Path("archive/2026-04")

# Copy a file (metadata preserved)
shutil.copy2("report.pdf", "backup/report.pdf")

# Copy an entire directory tree
shutil.copytree(src, dst)

# Move (works across filesystems, unlike Path.rename)
shutil.move("temp/output.csv", "final/output.csv")

# Delete a directory tree (careful!)
shutil.rmtree("build/", ignore_errors=True)

# Find the location of an executable (like `which` on Unix)
python_path = shutil.which("python3")
print(python_path)   # /usr/bin/python3

# Create a zip/tar archive
shutil.make_archive(
    base_name="backup-2026-04",   # output file name (no extension)
    format="zip",                  # or "tar", "gztar", "bztar"
    root_dir=".",
    base_dir="reports/",           # directory to archive
)
# Creates: backup-2026-04.zip

# Extract an archive
shutil.unpack_archive("backup-2026-04.zip", extract_dir="restored/")

3. schedule — Recurring Tasks

The schedule library runs functions on a timer with a readable API. It's the right choice for single-process scheduled tasks — simpler than Celery Beat when you don't need distributed workers.

$ pip install schedule
Python 
import schedule
import time
import logging

log = logging.getLogger(__name__)

def cleanup_temp_files():
    import shutil, pathlib
    tmp = pathlib.Path("/tmp/myapp")
    if tmp.exists():
        shutil.rmtree(tmp)
    log.info("Temp files cleaned")

def send_daily_report():
    log.info("Sending daily report...")
    # build and send report

def sync_data():
    log.info("Syncing data with external API...")

# Schedule declarations — very readable
schedule.every(30).minutes.do(cleanup_temp_files)
schedule.every().day.at("08:00").do(send_daily_report)
schedule.every().monday.at("09:00").do(sync_data)
schedule.every(5).seconds.do(lambda: log.debug("heartbeat"))

log.info("Scheduler started")
while True:
    schedule.run_pending()
    time.sleep(1)    # check every second
schedule vs Celery Beat: Use schedule for single-process scripts (a cron-replacement). Use Celery Beat when you need tasks distributed across workers, retries, or monitoring.

4. Sending Emails with smtplib

Python — plain text email 
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders
from pathlib import Path

def send_email(
    to: list[str],
    subject: str,
    body: str,
    attachments: list[Path] | None = None,
    smtp_host: str = "smtp.gmail.com",
    smtp_port: int = 587,
    from_addr: str = "",
    password: str = "",
) -> None:
    msg = MIMEMultipart()
    msg["From"] = from_addr
    msg["To"] = ", ".join(to)
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "plain"))

    # Attach files
    for filepath in (attachments or []):
        part = MIMEBase("application", "octet-stream")
        part.set_payload(filepath.read_bytes())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", f"attachment; filename={filepath.name}")
        msg.attach(part)

    with smtplib.SMTP(smtp_host, smtp_port) as server:
        server.starttls()               # upgrade to TLS
        server.login(from_addr, password)
        server.sendmail(from_addr, to, msg.as_string())

# Usage
import os
send_email(
    to=["boss@example.com"],
    subject="Daily Report — 2026-04-26",
    body="Please find today's report attached.",
    attachments=[Path("report.csv")],
    from_addr=os.environ["EMAIL_FROM"],
    password=os.environ["EMAIL_PASSWORD"],
)
Gmail tip: Use an App Password (not your real password) when authenticating with Gmail SMTP. Go to Google Account → Security → App passwords. Store it in an environment variable, never in code.

5. Real-world Automation: Report Generator

Combining everything — read data, generate a report, archive old files, send email, run on a schedule.

Python — daily_report.py 
import os
import shutil
import schedule
import time
import logging
from datetime import date
from pathlib import Path
import pandas as pd

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
log = logging.getLogger(__name__)

REPORT_DIR = Path("reports")
ARCHIVE_DIR = Path("archive")
REPORT_DIR.mkdir(exist_ok=True)
ARCHIVE_DIR.mkdir(exist_ok=True)

def generate_report() -> Path:
    """Pull data, build a CSV report, return the file path."""
    today = date.today().isoformat()
    output = REPORT_DIR / f"report-{today}.csv"

    # Simulate: replace with real DB/API query
    df = pd.DataFrame({
        "date": [today] * 3,
        "metric": ["sales", "signups", "churn"],
        "value": [12_450, 87, 3],
    })
    df.to_csv(output, index=False)
    log.info("report_generated", extra={"file": str(output)})
    return output

def archive_old_reports(days_old: int = 30) -> None:
    """Move reports older than N days to archive/."""
    cutoff = date.today().toordinal() - days_old
    for report in REPORT_DIR.glob("report-*.csv"):
        try:
            report_date = date.fromisoformat(report.stem.replace("report-", ""))
            if report_date.toordinal() < cutoff:
                shutil.move(str(report), ARCHIVE_DIR / report.name)
                log.info("report_archived", extra={"file": report.name})
        except ValueError:
            pass

def daily_job() -> None:
    log.info("Starting daily job")
    report_file = generate_report()
    archive_old_reports()
    # send_email(to=["team@example.com"], subject="Daily Report", ..., attachments=[report_file])
    log.info("Daily job complete")

schedule.every().day.at("07:00").do(daily_job)

if __name__ == "__main__":
    daily_job()   # run once immediately on start
    while True:
        schedule.run_pending()
        time.sleep(60)

Quick Reference

Run command
subprocess.run(["git", "status"], capture_output=True, text=True, check=True)

check=True raises on non-zero exit; text=True decodes bytes

Get stdout
result.stdout.strip()

Strip trailing newline; result.returncode for exit code

Copy file
shutil.copy2(src, dst)

copy2 preserves metadata; copy doesn't

Copy directory
shutil.copytree(src, dst)

Copies entire directory tree recursively

Create archive
shutil.make_archive("name", "zip", ".", "folder/")

Formats: zip, tar, gztar, bztar, xztar

Schedule
schedule.every().day.at("08:00").do(fn)

Also: .every(30).minutes, .every().monday

Run scheduler
while True: schedule.run_pending(); time.sleep(1)

Checks pending jobs every second; sleep(60) is fine for minute-resolution schedules

Send email
smtplib.SMTP("smtp.gmail.com", 587)

Call .starttls() before .login(); use App Password for Gmail

Wrap-up

  • Always pass commands as a list to subprocess.run() — never use shell=True with untrusted input. The list form is safer and doesn't require shell escaping.
  • shutil.make_archive() creates zip/tar archives in one line. This replaces 15–20 lines of Java ZipOutputStream code and is harder to get wrong.
  • schedule is the right choice for simple single-process recurring tasks. For tasks that need to survive process restarts, run on multiple machines, or retry on failure — use Celery Beat instead.
  • Store email credentials in environment variables, never in code. Use App Passwords for Gmail (not your account password) to enable SMTP access without lowering account security.
  • The real power of Python scripting is combining these tools: subprocess to run commands, pathlib/shutil for files, schedule for timing, smtplib for notifications — all in one readable script with no framework overhead.
