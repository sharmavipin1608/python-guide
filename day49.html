Project Spec

What You're Building

  • What: An installable Python package called tasklib. Core logic (models, storage, service) as a library. Click CLI as the user interface. Full pytest suite. pyproject.toml for packaging.
  • Package: tasklib — importable as from tasklib import TaskService
  • CLI: tasklib terminal command — add, list, done, delete sub-commands

  • Tests: pytest suite with 90%+ coverage; both unit and integration tests
  • Covers: Packaging (47), Click CLI (47), Design patterns (48), pytest (27), type hints (13), dataclasses (13)

Step 2 — Models + Exceptions

src/tasklib/models.py 
from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum

class Priority(str, Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"

class Status(str, Enum):
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    DONE = "done"

@dataclass
class Task:
    title: str
    id: int = 0
    priority: Priority = Priority.MEDIUM
    status: Status = Status.PENDING
    notes: str = ""
    created_at: datetime = field(
        default_factory=lambda: datetime.now(timezone.utc)
    )

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "title": self.title,
            "priority": self.priority.value,
            "status": self.status.value,
            "notes": self.notes,
            "created_at": self.created_at.isoformat(),
        }

    @classmethod
    def from_dict(cls, data: dict) -> "Task":
        return cls(
            id=data["id"],
            title=data["title"],
            priority=Priority(data["priority"]),
            status=Status(data["status"]),
            notes=data.get("notes", ""),
            created_at=datetime.fromisoformat(data["created_at"]),
        )

    """Base exception for all tasklib errors."""

class TaskNotFoundError(TaskLibError):
    def __init__(self, task_id: int) -> None:
        super().__init__(f"Task with id={task_id} not found")
        self.task_id = task_id

class StorageError(TaskLibError):
    """Raised when the storage backend fails."""

Step 3 — Storage Protocol + Implementations

src/tasklib/storage.py 
import json
import os
from typing import Protocol
from .models import Task, Status

class TaskStorage(Protocol):
    """Protocol — any class with these methods can be used as storage."""
    def save(self, task: Task) -> Task: ...
    def get(self, task_id: int) -> Task | None: ...
    def list_all(self, status: Status | None = None) -> list[Task]: ...
    def delete(self, task_id: int) -> None: ...

class JsonStorage:
    """File-backed JSON storage — good for the CLI."""

    def __init__(self, path: str = "~/.tasklib/tasks.json") -> None:
        self._path = os.path.expanduser(path)
        os.makedirs(os.path.dirname(self._path), exist_ok=True)

    def _load(self) -> dict[int, dict]:
        if not os.path.exists(self._path):
            return {}
        with open(self._path) as f:
            return {int(k): v for k, v in json.load(f).items()}

    def _dump(self, data: dict[int, dict]) -> None:
        with open(self._path, "w") as f:
            json.dump(data, f, indent=2)

    def save(self, task: Task) -> Task:
        data = self._load()
        if task.id == 0:
            task.id = max(data.keys(), default=0) + 1
        data[task.id] = task.to_dict()
        self._dump(data)
        return task

    def get(self, task_id: int) -> Task | None:
        return Task.from_dict(d) if (d := self._load().get(task_id)) else None

    def list_all(self, status: Status | None = None) -> list[Task]:
        tasks = [Task.from_dict(d) for d in self._load().values()]
        if status:
            tasks = [t for t in tasks if t.status == status]
        return sorted(tasks, key=lambda t: t.id)

    def delete(self, task_id: int) -> None:
        data = self._load()
        data.pop(task_id, None)
        self._dump(data)

class InMemoryStorage:
    """In-memory storage — perfect for tests."""

    def __init__(self) -> None:
        self._store: dict[int, Task] = {}
        self._next_id = 1

    def save(self, task: Task) -> Task:
        if task.id == 0:
            task.id = self._next_id
            self._next_id += 1
        self._store[task.id] = task
        return task

    def get(self, task_id: int) -> Task | None:
        return self._store.get(task_id)

    def list_all(self, status: Status | None = None) -> list[Task]:
        tasks = list(self._store.values())
        if status:
            tasks = [t for t in tasks if t.status == status]
        return sorted(tasks, key=lambda t: t.id)

    def delete(self, task_id: int) -> None:
        self._store.pop(task_id, None)

Step 4 — Service + Public API

src/tasklib/service.py 
from .models import Task, Priority, Status
from .storage import TaskStorage
from .exceptions import TaskNotFoundError

class TaskService:
    def __init__(self, storage: TaskStorage) -> None:
        self._storage = storage

    def create(self, title: str, priority: Priority = Priority.MEDIUM, notes: str = "") -> Task:
        if not title.strip():
            raise ValueError("Title cannot be empty")
        return self._storage.save(Task(title=title.strip(), priority=priority, notes=notes))

    def complete(self, task_id: int) -> Task:
        task = self._get_or_raise(task_id)
        task.status = Status.DONE
        return self._storage.save(task)

    def start(self, task_id: int) -> Task:
        task = self._get_or_raise(task_id)
        task.status = Status.IN_PROGRESS
        return self._storage.save(task)

    def delete(self, task_id: int) -> None:
        self._get_or_raise(task_id)
        self._storage.delete(task_id)

    def list_tasks(self, status: Status | None = None) -> list[Task]:
        return self._storage.list_all(status=status)

    def _get_or_raise(self, task_id: int) -> Task:
        task = self._storage.get(task_id)
        if not task:
            raise TaskNotFoundError(task_id)
        return task

Step 5 — Click CLI

src/tasklib/cli.py 
import click
from .models import Priority, Status
from .service import TaskService
from .storage import JsonStorage
from .exceptions import TaskNotFoundError

def get_service() -> TaskService:
    return TaskService(JsonStorage())

@click.group()
@click.version_option()
def main() -> None:
    """Tasklib — manage your tasks from the terminal."""

@main.command()
@click.argument("title")
@click.option("--priority", "-p",
              type=click.Choice([p.value for p in Priority]),
              default=Priority.MEDIUM.value, show_default=True)
@click.option("--notes", "-n", default="", help="Optional notes.")
def add(title: str, priority: str, notes: str) -> None:
    """Add a new task."""
    task = get_service().create(title, Priority(priority), notes)
    click.echo(click.style(f"✓ [{task.id}] {task.title}", fg="green"))

@main.command("list")
@click.option("--status", type=click.Choice([s.value for s in Status] + ["all"]), default="all")
def list_tasks(status: str) -> None:
    """List tasks."""
    svc = get_service()
    tasks = svc.list_tasks(Status(status) if status != "all" else None)
    if not tasks:
        click.echo("No tasks.")
        return
    for t in tasks:
        icon  = "✓" if t.status == Status.DONE else ("→" if t.status == Status.IN_PROGRESS else "○")
        color = "green" if t.status == Status.DONE else ("cyan" if t.status == Status.IN_PROGRESS else "white")
        click.echo(click.style(f"{icon} [{t.id}] {t.title} ({t.priority.value})", fg=color))

@main.command()
@click.argument("task_id", type=int)
def done(task_id: int) -> None:
    """Mark a task as done."""
    try:
        task = get_service().complete(task_id)
        click.echo(click.style(f"✓ Completed: {task.title}", fg="green"))
    except TaskNotFoundError as e:
        click.echo(click.style(str(e), fg="red"), err=True)
        raise SystemExit(1)

@main.command()
@click.argument("task_id", type=int)
@click.confirmation_option(prompt="Delete this task?")
def delete(task_id: int) -> None:
    """Delete a task."""
    try:
        get_service().delete(task_id)
        click.echo(f"Deleted task {task_id}.")
    except TaskNotFoundError as e:
        click.echo(click.style(str(e), fg="red"), err=True)
        raise SystemExit(1)

tests/test_service.py 
import pytest
from tasklib import TaskService, InMemoryStorage, Priority, Status
from tasklib.exceptions import TaskNotFoundError

@pytest.fixture
def service() -> TaskService:
    return TaskService(InMemoryStorage())

def test_create_task(service: TaskService) -> None:
    task = service.create("Buy milk")
    assert task.id > 0
    assert task.title == "Buy milk"
    assert task.status == Status.PENDING

def test_create_assigns_incremental_ids(service: TaskService) -> None:
    t1 = service.create("Task 1")
    t2 = service.create("Task 2")
    assert t2.id == t1.id + 1

def test_create_empty_title_raises(service: TaskService) -> None:
    with pytest.raises(ValueError, match="empty"):
        service.create("   ")

def test_complete_task(service: TaskService) -> None:
    task = service.create("Write tests")
    service.complete(task.id)
    updated = service.list_tasks(Status.DONE)
    assert any(t.id == task.id for t in updated)

def test_complete_nonexistent_raises(service: TaskService) -> None:
    with pytest.raises(TaskNotFoundError):
        service.complete(9999)

def test_list_filters_by_status(service: TaskService) -> None:
    t1 = service.create("Task 1")
    t2 = service.create("Task 2")
    service.complete(t1.id)
    pending = service.list_tasks(Status.PENDING)
    assert all(t.status == Status.PENDING for t in pending)
    assert any(t.id == t2.id for t in pending)

def test_delete_task(service: TaskService) -> None:
    task = service.create("To delete")
    service.delete(task.id)
    assert service.list_tasks() == []

from click.testing import CliRunner
from tasklib.cli import main
from tasklib import InMemoryStorage, TaskService
import tasklib.cli as cli_module
import pytest

@pytest.fixture(autouse=True)
def patch_storage(monkeypatch):
    """Replace JsonStorage with InMemoryStorage in tests."""
    storage = InMemoryStorage()

def test_add_command():

Wrap-up

  • The Protocol-based storage layer is the key architectural decision: InMemoryStorage makes tests instant and DB-free; JsonStorage works for CLI; you could add SQLAlchemyStorage for the API — zero changes to TaskService.
  • Expose a clean __init__.py with explicit __all__. Users should be able to do from tasklib import TaskService without knowing the internal module structure.
  • Click's CliRunner is the right tool for CLI tests — it captures stdout, stderr, and exit codes without spawning real processes. Combine it with monkeypatch to swap real I/O for fast in-memory implementations.
  • mypy --strict on the whole package at this stage is achievable and worthwhile — you have full type hints and a proper package structure. Fix the mypy errors before publishing.
  • The pyproject.toml tool configuration section ([tool.mypy], [tool.ruff], [tool.pytest.ini_options]) replaces separate config files. One file to rule them all.
