Day 42 — Project Phase 3 — Backend + AI

Project: Mini RAG Chatbot

Upload documents, ask questions, get answers with source references. This project wires together everything from Phase 3: FastAPI, LangChain, ChromaDB, the OpenAI SDK, async, and structured logging.

Project Spec

What You're Building

  • What: A FastAPI service for document Q&A. Upload text or PDF files → they're chunked and indexed in ChromaDB → ask questions → get answers with source references.
  • Endpoints: POST /documents (upload + index), GET /documents (list indexed docs), POST /chat (ask a question), DELETE /documents/{id} (remove from index)
  • Storage: ChromaDB (persistent, on disk) for vectors; in-memory dict for document metadata
  • LLM: OpenAI gpt-4o-mini (or swap for ChatOllama for local use)
  • Covers: FastAPI (29–30), ChromaDB (38), OpenAI SDK (39), LangChain (36), async, structlog (41)
Stretch Goal: Add conversation memory so follow-up questions work. Store session_id → message_history in memory and pass the last N messages as context on each question. Also: add a streaming endpoint (POST /chat/stream) that returns tokens via SSE.

from pydantic import BaseModel

class DocumentInfo(BaseModel):
    doc_id: str
    filename: str
    chunk_count: int

    question: str
    session_id: str = "default"

class Source(BaseModel):
    doc_id: str
    filename: str

class AnswerResponse(BaseModel):
    answer: str
    sources: list[Source]

Step 2 — Indexer

indexer.py 
import uuid
import structlog
import chromadb
from chromadb.utils.embedding_functions import OpenAIEmbeddingFunction
from langchain_text_splitters import RecursiveCharacterTextSplitter
from config import get_settings

log = structlog.get_logger(__name__)
settings = get_settings()

def get_collection() -> chromadb.Collection:
    client = chromadb.PersistentClient(path=settings.chroma_path)
    embedding_fn = OpenAIEmbeddingFunction(
        api_key=settings.openai_api_key,
        model_name="text-embedding-3-small",
    )
    return client.get_or_create_collection(
        name=settings.chroma_collection,
        embedding_function=embedding_fn,
    )

def index_document(filename: str, content: str) -> tuple[str, int]:
    """Chunk and index a document. Returns (doc_id, chunk_count)."""
    doc_id = str(uuid.uuid4())
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=settings.chunk_size,
        chunk_overlap=settings.chunk_overlap,
    )
    chunks = splitter.split_text(content)
    if not chunks:
        raise ValueError("Document produced no chunks after splitting")

    collection = get_collection()
    collection.add(
        documents=chunks,
        ids=[f"{doc_id}-chunk-{i}" for i in range(len(chunks))],
        metadatas=[{"doc_id": doc_id, "filename": filename, "chunk_index": i}
                   for i in range(len(chunks))],
    )
    log.info("document_indexed", doc_id=doc_id, filename=filename, chunks=len(chunks))
    return doc_id, len(chunks)

def delete_document(doc_id: str) -> int:
    """Delete all chunks belonging to a document. Returns chunks deleted."""
    collection = get_collection()
    results = collection.get(where={"doc_id": doc_id})
    ids = results["ids"]
    if ids:
        collection.delete(ids=ids)
    log.info("document_deleted", doc_id=doc_id, chunks_removed=len(ids))
    return len(ids)

Step 3 — Retriever + RAG Chain

retriever.py 
from indexer import get_collection
from models import Source
from config import get_settings

settings = get_settings()

def retrieve(question: str) -> tuple[str, list[Source]]:
    """Find the most relevant chunks for a question."""
    collection = get_collection()
    results = collection.query(
        query_texts=[question],
        n_results=settings.retrieval_k,
    )
    docs = results["documents"][0]
    metas = results["metadatas"][0]

    context = "\n\n".join(
        f"[Source {i+1} — {m['filename']}]:\n{doc}"
        for i, (doc, m) in enumerate(zip(docs, metas))
    )
    sources = [
        Source(
            doc_id=m["doc_id"],
            filename=m["filename"],
            chunk_preview=doc[:100],
        )
        for doc, m in zip(docs, metas)
    ]
    return context, sources
chatbot.py 
import structlog
from openai import AsyncOpenAI
from retriever import retrieve
from models import AnswerResponse
from config import get_settings

log = structlog.get_logger(__name__)
settings = get_settings()

SYSTEM_PROMPT = """You are a helpful assistant that answers questions based on provided documents.
Answer using ONLY the information in the context below.
If the context doesn't contain the answer, say: "I don't have enough information to answer that."
Always be concise and cite which source you're drawing from.

Context:
{context}"""

async def answer_question(question: str) -> AnswerResponse:
    log.info("question_received", question=question[:100])

    # Retrieve relevant chunks
    context, sources = retrieve(question)

    # Generate answer
    client = AsyncOpenAI(api_key=settings.openai_api_key)
    response = await client.chat.completions.create(
        model=settings.openai_model,
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT.format(context=context)},
            {"role": "user", "content": question},
        ],
        temperature=0,
        max_tokens=600,
    )
    answer = response.choices[0].message.content
    log.info("answer_generated", sources=len(sources), tokens=response.usage.total_tokens)

    return AnswerResponse(answer=answer, sources=sources)

Step 4 — FastAPI App

main.py 
import sys
import structlog
from fastapi import FastAPI, UploadFile, File, HTTPException
from models import DocumentInfo, QuestionRequest, AnswerResponse
from indexer import index_document, delete_document
from chatbot import answer_question

# Logging setup
structlog.configure(
    processors=[
        structlog.contextvars.merge_contextvars,
        structlog.stdlib.add_log_level,
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.dev.ConsoleRenderer(),   # switch to JSONRenderer() in prod
    ],
    logger_factory=structlog.stdlib.LoggerFactory(),
)

log = structlog.get_logger(__name__)

# In-memory document registry (use a DB for production)
doc_registry: dict[str, DocumentInfo] = {}

app = FastAPI(title="RAG Chatbot", version="1.0")

@app.post("/documents", response_model=DocumentInfo, status_code=201)
async def upload_document(file: UploadFile = File(...)):
    """Upload and index a document (plain text or PDF)."""
    if file.content_type not in ("text/plain", "application/pdf"):
        raise HTTPException(status_code=400, detail="Only .txt and .pdf files are supported")

    content_bytes = await file.read()
    if file.content_type == "application/pdf":
        from pypdf import PdfReader
        import io
        reader = PdfReader(io.BytesIO(content_bytes))
        content = "\n".join(page.extract_text() or "" for page in reader.pages)
    else:
        content = content_bytes.decode("utf-8", errors="ignore")

    if not content.strip():
        raise HTTPException(status_code=422, detail="Document appears to be empty")

    doc_id, chunk_count = index_document(file.filename, content)
    info = DocumentInfo(doc_id=doc_id, filename=file.filename, chunk_count=chunk_count)
    doc_registry[doc_id] = info
    return info

@app.get("/documents", response_model=list[DocumentInfo])
def list_documents():
    return list(doc_registry.values())

@app.delete("/documents/{doc_id}", status_code=204)
def remove_document(doc_id: str):
    if doc_id not in doc_registry:
        raise HTTPException(status_code=404, detail="Document not found")
    delete_document(doc_id)
    del doc_registry[doc_id]

@app.post("/chat", response_model=AnswerResponse)
async def chat(request: QuestionRequest):
    """Ask a question about the indexed documents."""
    if not request.question.strip():
        raise HTTPException(status_code=400, detail="Question cannot be empty")
    structlog.contextvars.bind_contextvars(session_id=request.session_id)
    return await answer_question(request.question)

@app.get("/health")
def health():
    return {"status": "ok", "documents_indexed": len(doc_registry)}

Stretch: Conversation Memory

chatbot.py — with session memory 
from collections import defaultdict
from openai.types.chat import ChatCompletionMessageParam

# Session store: session_id → list of messages
_sessions: dict[str, list[ChatCompletionMessageParam]] = defaultdict(list)

async def answer_with_memory(question: str, session_id: str) -> AnswerResponse:
    context, sources = retrieve(question)
    history = _sessions[session_id]

    messages: list[ChatCompletionMessageParam] = [
        {"role": "system", "content": SYSTEM_PROMPT.format(context=context)},
        *history[-6:],                      # last 3 turns (6 messages)
        {"role": "user", "content": question},
    ]

    client = AsyncOpenAI(api_key=settings.openai_api_key)
    response = await client.chat.completions.create(
        model=settings.openai_model,
        messages=messages,
        temperature=0,
    )
    answer = response.choices[0].message.content

    # Update history
    _sessions[session_id].append({"role": "user", "content": question})
    _sessions[session_id].append({"role": "assistant", "content": answer})

    return AnswerResponse(answer=answer, sources=sources)

Wrap-up

  • Separating indexer, retriever, and chatbot into distinct modules makes the system testable — you can unit-test retrieval quality without an LLM, and test the LLM prompt without ChromaDB.
  • Metadata on ChromaDB documents (doc_id, filename) is what makes deletion and source attribution possible. Always store enough metadata to answer "which document did this come from?"
  • The system prompt boundary — "answer ONLY using the context" — is critical for preventing hallucination. Without it, the LLM will make things up when it doesn't find the answer in the retrieved chunks.
  • Session memory is a sliding window, not an unlimited history. Six messages (3 turns) is a practical limit that keeps context costs controlled while preserving conversational coherence.
  • This is the full RAG architecture used in production AI products — vector store, retrieval, augmented prompt, generation. The main difference at scale is using a managed vector DB (Pinecone, Weaviate) and a dedicated embedding pipeline.
